79 /* The pass() function is the low-level core of the testsuite.
80  *
81  * You call this with two arguments ... an integer expression indicating the
82  * success or failure of the testcase (0 is a failure) and a string which
83  * describes the testcase.
84  *
85  * The global variable 'testHopeful' can be set to a non-zero value before
86  * calling this function in order to specify that if the condition is
87  * not true it should be treated as a dashed hope rather than a failure
88  * (unless the tests are bing performed in 'developer' mode).
89  *
90  * If there is a better higher-level test macro available, please use
91  * that instead. In particular, please use the PASS_EQUAL() macro wherever
92  * you wish to test the equality of a pair of objective-c objects.
93  *
94  * If you are calling the function directly, please use a format string
95  * beginning "%s:%d" and pass __FILE__ and __LINE__ as the first two arguments
96  * so that this function will print out the location it was called from.
97  *
98  * This function is the most efficient option for general use, but
99  * please don't use it if there is any change that the evaluation of
100  * the expression used as its first argument might cause an exception
101  * in any context where that might be a problem.
102  */
103 static void pass(int passed, const char *format, ...) __attribute__((unused)) __attribute__ ((format(printf, 2, 3)));
104 static void pass(int passed, const char *format, ...)
105 {
106  va_list args;
107  va_start(args, format);
108 
109  if (passed)
110  {
111  fprintf(stderr, "Passed test: ");
112  testPassed = YES;
113  }
114 #if !defined(TESTDEV)
115  else if (YES == testHopeful)
116  {
117  fprintf(stderr, "Dashed hope: ");
118  testPassed = NO;
119  }
120 #endif
121  else
122  {
123  fprintf(stderr, "Failed test: ");
124  testPassed = NO;
125  }
126  testIndent();
127  vfprintf(stderr, format, args);
128  fprintf(stderr, "\n");
129  va_end(args);
130 #if defined(FAILFAST)
131  if (NO == testPassed && NO == testHopeful)
132  {
133  exit(1); // Abandon testing now.
134  }
135 #endif
136 }

151 /* Tests a code expression which evaluates to an integer value.
152  * The expression may not contain commas unless it is bracketed.
153  * The format must be a literal string printf style format.
154  * If the expression evaluates to zero the test does not pass.
155  * If the expression causes an exception to be raised, the exception
156  * is caught and logged but the test does not pass.
157  * Otherwise, the test passes.
158  * Basically equivalent to pass() but with exception handling.
159  */
160 #define PASS(testExpression__, testFormat__, ...) \
161  NS_DURING \
162  { \
163  int _cond; \
164  id _tmp = testRaised; testRaised = nil; [_tmp release]; \
165  [[NSGarbageCollector defaultCollector] collectExhaustively]; \
166  testLineNumber = __LINE__; \
167  testStart(); \
168  _cond = (int)(testExpression__); \
169  [[NSGarbageCollector defaultCollector] collectExhaustively]; \
170  pass(_cond, "%s:%d ... " testFormat__, __FILE__, \
171  __LINE__, ## __VA_ARGS__); \
172  } \
173  NS_HANDLER \
174  testRaised = [localException retain]; \
175  pass(0, "%s:%d ... " testFormat__, __FILE__, __LINE__, ## __VA_ARGS__); \
176  printf("%s: %s", [[testRaised name] UTF8String], \
177  [[testRaised description] UTF8String]); \
178  NS_ENDHANDLER

180 /* Tests a code expression which evaluates to an object value.
181  * The expression may not contain commas unless it is bracketed.
182  * The expected value may not contain commas unless it is bracketed.
183  * The format must be a literal string printf style format.
184  *
185  * Where the expression evaluates to an object which is identical to
186  * the expect value, or where [object isEqual: expect] returns YES,
187  * the test has passed.
188  *
189  * The particularly useful thing about this macro is that, if the
190  * results of the expression and the expected object are not equal,
191  * the string representation of both values is logged so that you
192  * can get a better idea of what went wrong.
193  */
194 #define PASS_EQUAL(testExpression__, testExpect__, testFormat__, ...) \
195  NS_DURING \
196  { \
197  int _cond; \
198  id _obj; \
199  id _exp; \
200  id _tmp = testRaised; testRaised = nil; [_tmp release]; \
201  [[NSGarbageCollector defaultCollector] collectExhaustively]; \
202  testLineNumber = __LINE__; \
203  testStart(); \
204  _obj = (id)(testExpression__);\
205  _exp = (id)(testExpect__);\
206  _cond = _obj == _exp || [_exp isEqual: _obj]; \
207  [[NSGarbageCollector defaultCollector] collectExhaustively]; \
208  pass(_cond, "%s:%d ... " testFormat__, __FILE__, __LINE__, ## __VA_ARGS__); \
209  if (0 == _cond) \
210  { \
211  NSString *s = [_obj description]; \
212  if ([s length] == 1) \
213  { \
214  fprintf(stderr, \
215  "Expected '%s' and got '%s' (unicode codepoint %d)\n", \
216  [[_exp description] UTF8String], [s UTF8String], \
217  [s characterAtIndex: 0]); \
218  } \
219  else if (nil == s) \
220  { \
221  fprintf(stderr, "Expected '%s' and got (nil)\n", \
222  [[_exp description] UTF8String]); \
223  } \
224  else \
225  { \
226  fprintf(stderr, "Expected '%s' and got '%s'\n", \
227  [[_exp description] UTF8String], [s UTF8String]); \
228  } \
229  } \

293 /* SETs are used to group multiple testcases or code which is outside of
294  * the scope of the current test but could raise exceptions that should
295  * be caught to allow further tests to run.
296  *
297  * You must pass a short description to identify the set at both its
298  * start and its end. This allows the seat to be easily identified in the
299  * log, and also allows for checking to be sure that each start if a set
300  * is matched by a corresponding end.
301  *
302  * The state of the 'testHopeful' flag is saved at the start of the set and
303  * restored at the end of the set, so you can start your code by setting
304  * 'testHopeful=YES;' to mark any tests within the set as being part of a
305  * group of tests we don't expect to pass.
306  *
307  * Importantly, you may skip some or all of the tests in a set if those
308  * tests are not supported in the package being tested (eg. testing of
309  * functionality which depends on some external library which was not
310  * available when the package being tested was buit).
311  *
312  * Any uncaught exception occurring inside a set will abort the entire set
313  * so that remaining tests in the set will not be executed, but you may
314  * also abandon remaining tests upon any test failure.
315  *
316  * The tests within the set are enclosed in an autorelease pool, and any
317  * temporary objects are cleaned up at the end of the set.
318  */

320 /* The START_SET() macro starts a set of grouped tests. It must be matched
321  * by a corresponding END_SET() with the same string as an argument.
322  * The argument is a short description to be printed in the log on entry.
323  */
324 #define START_SET(setName) \
325  { \
326  BOOL _save_hopeful = testHopeful; \
327  unsigned _save_indentation = testIndentation; \
328  int _save_line = __LINE__; \
329  char *_save_set = (char*)malloc(strlen(setName) + 1); \
330  strncpy(_save_set, setName, strlen(setName) + 1); \
331  fprintf(stderr, "Start set: "); \
332  testIndent(); \
333  fprintf(stderr, "%s:%d ... %s\n", __FILE__, __LINE__, _save_set); \
334  testIndentation++; \
335  NS_DURING \
336  NSAutoreleasePool *_setPool = [NSAutoreleasePool new]; \
337  {

352 /* The END_SET() macro terminates a set of grouped tests. It must be matched
353  * by a corresponding START_SET() with the same string as an argument.
354  * The argument is a short description to be printed in the log on entry.
355  */
356 #define END_SET(setName) \
357  } \

371  { \

383  fprintf(stderr, "Failed set: "); \
384  testIndent(); \
385  fprintf(stderr, "%s:%d ... problem in %s.\n", \
386  __FILE__, _save_line, _save_set); \
387  } \
388  } \
389  NS_ENDHANDLER \
390  if (strcmp(_save_set, setName) != 0) \
391  fprintf(stderr, "Error: %s:%d ... END(%s) with START(%s).\n", \
392  __FILE__, __LINE__, setName, _save_set); \
393  testIndentation = _save_indentation; \
394  fprintf(stderr, "End set: "); \
395  testIndent(); \
396  fprintf(stderr, "%s:%d ... %s\n", __FILE__, __LINE__, _save_set); \
397  free(_save_set); \
398  testHopeful = _save_hopeful; \
399  }
